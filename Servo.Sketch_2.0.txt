#include <Servo.h>

Servo myservo;
int pos = 0;
int sensorpin = A0;
void setup() 
{
Serial.begin(9600);
pinMode(sensorpin, INPUT); 
myservo.attach(9);
}
 
void loop() 
{
int pos;
myservo.write(0);
if(analogRead(sensorpin)> 0)
{
 for (pos = 0; pos <= 180; pos=pos +1) 
{
  myservo.write(pos);
}
if(analogRead(sensorpin)== 0)
{
for(pos=180; pos >= 0; pos=pos-1)
{
myservo.write(pos);
}
}
}
}
